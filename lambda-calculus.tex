  \vspace{-6mm}

  \[ x \in V \Rightarrow x \in \Lambda \]
  \[ M, N \in \Lambda \Rightarrow (M \  N) \in \Lambda \]
  \[ M \in \Lambda, x \in V \Rightarrow (\lambda x.M ) \in \Lambda \]

  { \color{gmitblue}

        \end{tabular}
      \end{table}
    \end{center}
  }
